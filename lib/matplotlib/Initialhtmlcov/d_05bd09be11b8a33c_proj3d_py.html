11def _line2d_seg_dist(p, s0, s1): 

12 """ 

13 Return the distance(s) from point(s) *p* to segment(s) (*s0*, *s1*). 

14 

15 Parameters 

16 ---------- 

17 p : (ndim,) or (N, ndim) array-like 

18 The points from which the distances are computed. 

19 s0, s1 : (ndim,) or (N, ndim) array-like 

20 The xy(z...) coordinates of the segment endpoints. 

21 """ 

22 s0 = np.asarray(s0) 

23 s01 = s1 - s0 # shape (ndim,) or (N, ndim) 

24 s0p = p - s0 # shape (ndim,) or (N, ndim) 

25 l2 = s01 @ s01 # squared segment length 

26 # Avoid div. by zero for degenerate segments (for them, s01 = (0, 0, ...) 

27 # so the value of l2 doesn't matter; this just replaces 0/0 by 0/1). 

28 l2 = np.where(l2, l2, 1) 

29 # Project onto segment, without going past segment ends. 

30 p1 = s0 + np.multiply.outer(np.clip(s0p @ s01 / l2, 0, 1), s01) 

31 return ((p - p1) ** 2).sum(axis=-1) ** (1/2)

64def _rotation_about_vector(v, angle): 

65 """ 

66 Produce a rotation matrix for an angle in radians about a vector. 

67 """ 

68 vx, vy, vz = v / np.linalg.norm(v) 

69 s = np.sin(angle) 

70 c = np.cos(angle) 

71 t = 2*np.sin(angle/2)**2 # more numerically stable than t = 1-c 

72 

73 R = np.array([ 

74 [t*vx*vx + c, t*vx*vy - vz*s, t*vx*vz + vy*s], 

75 [t*vy*vx + vz*s, t*vy*vy + c, t*vy*vz - vx*s], 

76 [t*vz*vx - vy*s, t*vz*vy + vx*s, t*vz*vz + c]]) 

77 

78 return R

81def _view_axes(E, R, V, roll): 

82 """ 

83 Get the unit viewing axes in data coordinates. 

84 

85 Parameters 

86 ---------- 

87 E : 3-element numpy array 

88 The coordinates of the eye/camera. 

89 R : 3-element numpy array 

90 The coordinates of the center of the view box. 

91 V : 3-element numpy array 

92 Unit vector in the direction of the vertical axis. 

93 roll : float 

94 The roll angle in radians. 

95 

96 Returns 

97 ------- 

98 u : 3-element numpy array 

99 Unit vector pointing towards the right of the screen. 

100 v : 3-element numpy array 

101 Unit vector pointing towards the top of the screen. 

102 w : 3-element numpy array 

103 Unit vector pointing out of the screen. 

104 """ 

105 w = (E - R) 

106 w = w/np.linalg.norm(w) 

107 u = np.cross(V, w) 

108 u = u/np.linalg.norm(u) 

109 v = np.cross(w, u) # Will be a unit vector 

110 

111 # Save some computation for the default roll=0 

112 if roll != 0: 112 ↛ 114line 112 didn't jump to line 114, because the condition on line 112 was never true

113 # A positive rotation of the camera is a negative rotation of the world 

114 Rroll = _rotation_about_vector(w, -roll) 

115 u = np.dot(Rroll, u) 

116 v = np.dot(Rroll, v) 

117 return u, v, w

120def _view_transformation_uvw(u, v, w, E): 

121 """ 

122 Return the view transformation matrix. 

123 

124 Parameters 

125 ---------- 

126 u : 3-element numpy array 

127 Unit vector pointing towards the right of the screen. 

128 v : 3-element numpy array 

129 Unit vector pointing towards the top of the screen. 

130 w : 3-element numpy array 

131 Unit vector pointing out of the screen. 

132 E : 3-element numpy array 

133 The coordinates of the eye/camera. 

134 """ 

135 Mr = np.eye(4) 

136 Mt = np.eye(4) 

137 Mr[:3, :3] = [u, v, w] 

138 Mt[:3, -1] = -E 

139 M = np.dot(Mr, Mt) 

140 return M
