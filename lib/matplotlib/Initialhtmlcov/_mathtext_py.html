104 def to_vector(self): 

105 w, h, d = map( 

106 np.ceil, [self.box.width, self.box.height, self.box.depth]) 

107 gs = [(info.font, info.fontsize, info.num, ox, h - oy + info.offset) 

108 for ox, oy, info in self.glyphs]

113 def to_raster(self): 

114 # Metrics y's and mathtext y's are oriented in opposite directions, 

115 # hence the switch between ymin and ymax. 

116 xmin = min([*[ox + info.metrics.xmin for ox, oy, info in self.glyphs], 

117 *[x1 for x1, y1, x2, y2 in self.rects], 0]) - 1 

118 ymin = min([*[oy - info.metrics.ymax for ox, oy, info in self.glyphs], 

119 *[y1 for x1, y1, x2, y2 in self.rects], 0]) - 1 

120 xmax = max([*[ox + info.metrics.xmax for ox, oy, info in self.glyphs], 

121 *[x2 for x1, y1, x2, y2 in self.rects], 0]) + 1 

122 ymax = max([*[oy - info.metrics.ymin for ox, oy, info in self.glyphs], 

123 *[y2 for x1, y1, x2, y2 in self.rects], 0]) + 1 

124 w = xmax - xmin 

125 h = ymax - ymin - self.box.depth 

126 d = ymax - ymin - self.box.height 

127 image = FT2Image(np.ceil(w), np.ceil(h + max(d, 0))) 

128 

129 # Ideally, we could just use self.glyphs and self.rects here, shifting 

130 # their coordinates by (-xmin, -ymin), but this yields slightly 

131 # different results due to floating point slop; shipping twice is the 

132 # old approach and keeps baseline images backcompat. 

133 shifted = ship(self.box, (-xmin, -ymin)) 

134 

135 for ox, oy, info in shifted.glyphs: 

136 info.font.draw_glyph_to_bitmap( 

137 image, ox, oy - info.metrics.iceberg, info.glyph, 

138 antialiased=mpl.rcParams['text.antialiased'])

146 image.draw_rect_filled(int(x1), y, np.ceil(x2), y + height) 

147 return RasterParse(0, 0, w, h + d, d, image)
