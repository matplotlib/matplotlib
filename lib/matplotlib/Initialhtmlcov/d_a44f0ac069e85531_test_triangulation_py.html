540 """

543 provided.

564 diff_dzy = tab_dzy - dzy

611 # Now 3 basic tests of the Sparse CG solver, used for 

612 # TriCubicInterpolator with *kind* = 'min_E' 

613 # 1) A commonly used test involves a 2d Poisson matrix. 

614 def poisson_sparse_matrix(n, m): 

615 """ 

616 Return the sparse, (n*m, n*m) matrix in coo format resulting from the 

617 discretisation of the 2-dimensional Poisson equation according to a 

618 finite difference numerical scheme on a uniform (n, m) grid. 

619 """ 

620 l = m*n 

621 rows = np.concatenate([ 

622 np.arange(l, dtype=np.int32), 

623 np.arange(l-1, dtype=np.int32), np.arange(1, l, dtype=np.int32), 

624 np.arange(l-n, dtype=np.int32), np.arange(n, l, dtype=np.int32)]) 

625 cols = np.concatenate([ 

626 np.arange(l, dtype=np.int32), 

627 np.arange(1, l, dtype=np.int32), np.arange(l-1, dtype=np.int32), 

628 np.arange(n, l, dtype=np.int32), np.arange(l-n, dtype=np.int32)]) 

629 vals = np.concatenate([ 

630 4*np.ones(l, dtype=np.float64), 

631 -np.ones(l-1, dtype=np.float64), -np.ones(l-1, dtype=np.float64), 

632 -np.ones(l-n, dtype=np.float64), -np.ones(l-n, dtype=np.float64)]) 

633 # In fact +1 and -1 diags have some zeros 

634 vals[l:2*l-1][m-1::m] = 0. 

635 vals[2*l-1:3*l-2][m-1::m] = 0. 

636 return vals, rows, cols, (n*m, n*m)

639 (n, m) = (12, 4)

646 b[itest] = 1.

652 # (but still linked with the rest of the matrix by extra-diag terms)

658 rows = np.concatenate([rows, [i_zero, i_zero-1, j_zero, j_zero-1]]) 

659 cols = np.concatenate([cols, [i_zero-1, i_zero, j_zero-1, j_zero]])

1091def meshgrid_triangles(n): 

1092 """ 

1093 Return (2*(N-1)**2, 3) array of triangles to mesh (N, N)-point np.meshgrid. 

1094 """ 

1095 tri = [] 

1096 for i in range(n-1): 

1097 for j in range(n-1): 

1098 a = i + j*n 

1099 b = (i+1) + j*n 

1100 c = i + (j+1)*n 

1101 d = (i+1) + (j+1)*n 

1102 tri += [[a, b, d], [a, d, c]] 

1103 return np.array(tri, dtype=np.int32)
