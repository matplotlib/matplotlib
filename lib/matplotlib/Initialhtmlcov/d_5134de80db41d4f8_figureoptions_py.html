30def figure_edit(axes, parent=None): 

31 """Edit matplotlib figure options""" 

32 sep = (None, None) # separator 

33 

34 # Get / General 

35 def convert_limits(lim, converter): 

36 """Convert axis limits for correct input editors.""" 

37 if isinstance(converter, DateConverter): 

38 return map(num2date, lim) 

39 # Cast to builtin floats as they have nicer reprs. 

40 return map(float, lim) 

41 

42 axis_map = axes._axis_map 

43 axis_limits = { 

44 name: tuple(convert_limits( 

45 getattr(axes, f'get_{name}lim')(), axis.converter 

46 )) 

47 for name, axis in axis_map.items() 

48 } 

49 general = [ 

50 ('Title', axes.get_title()), 

51 sep, 

52 *chain.from_iterable([ 

53 ( 

54 (None, f"<b>{name.title()}-Axis</b>"), 

55 ('Min', axis_limits[name][0]), 

56 ('Max', axis_limits[name][1]), 

57 ('Label', axis.get_label().get_text()), 

58 ('Scale', [axis.get_scale(), 

59 'linear', 'log', 'symlog', 'logit']), 

60 sep, 

61 ) 

62 for name, axis in axis_map.items() 

63 ]), 

64 ('(Re-)Generate automatic legend', False), 

65 ] 

66 

67 # Save the converter and unit data 

68 axis_converter = { 

69 name: axis.converter 

70 for name, axis in axis_map.items() 

71 } 

72 axis_units = { 

73 name: axis.get_units() 

74 for name, axis in axis_map.items() 

75 } 

76 

77 # Get / Curves 

78 labeled_lines = [] 

79 for line in axes.get_lines(): 

80 label = line.get_label() 

81 if label == '_nolegend_': 

82 continue 

83 labeled_lines.append((label, line)) 

84 curves = []

86 def prepare_data(d, init): 

87 """ 

88 Prepare entry for FormLayout. 

89 

90 *d* is a mapping of shorthands to style names (a single style may 

91 have multiple shorthands, in particular the shorthands `None`, 

92 `"None"`, `"none"` and `""` are synonyms); *init* is one shorthand 

93 of the initial style. 

94 

95 This function returns an list suitable for initializing a 

96 FormLayout combobox, namely `[initial_name, (shorthand, 

97 style_name), (shorthand, style_name), ...]`. 

98 """ 

99 if init not in d: 

100 d = {**d, init: str(init)} 

101 # Drop duplicate shorthands from dict (by overwriting them during 

102 # the dict comprehension). 

103 name2short = {name: short for short, name in d.items()} 

104 # Convert back to {shorthand: name}. 

105 short2name = {short: name for name, short in name2short.items()} 

106 # Find the kept shorthand for the style specified by init. 

107 canonical_init = name2short[d[init]] 

108 # Sort by representation and prepend the initial value. 

109 return ([canonical_init] + 

110 sorted(short2name.items(), 

111 key=lambda short_and_name: short_and_name[1])) 

112 

113 for label, line in labeled_lines: 

114 color = mcolors.to_hex( 

115 mcolors.to_rgba(line.get_color(), line.get_alpha()), 

116 keep_alpha=True) 

117 ec = mcolors.to_hex( 

118 mcolors.to_rgba(line.get_markeredgecolor(), line.get_alpha()), 

119 keep_alpha=True) 

120 fc = mcolors.to_hex( 

121 mcolors.to_rgba(line.get_markerfacecolor(), line.get_alpha()), 

122 keep_alpha=True) 

123 curvedata = [ 

124 ('Label', label), 

125 sep, 

126 (None, '<b>Line</b>'), 

127 ('Line style', prepare_data(LINESTYLES, line.get_linestyle())), 

128 ('Draw style', prepare_data(DRAWSTYLES, line.get_drawstyle())), 

129 ('Width', line.get_linewidth()), 

130 ('Color (RGBA)', color), 

131 sep, 

132 (None, '<b>Marker</b>'), 

133 ('Style', prepare_data(MARKERS, line.get_marker())), 

134 ('Size', line.get_markersize()), 

135 ('Face color (RGBA)', fc), 

136 ('Edge color (RGBA)', ec)] 

137 curves.append([curvedata, label, ""]) 

138 # Is there a curve displayed? 

139 has_curve = bool(curves) 

140 

141 # Get ScalarMappables. 

142 labeled_mappables = [] 

143 for mappable in [*axes.images, *axes.collections]: 

144 label = mappable.get_label() 

145 if label == '_nolegend_' or mappable.get_array() is None: 

146 continue 

147 labeled_mappables.append((label, mappable)) 

148 mappables = [] 

149 cmaps = [(cmap, name) for name, cmap in sorted(cm._colormaps.items())] 

150 for label, mappable in labeled_mappables: 

151 cmap = mappable.get_cmap() 

152 if cmap not in cm._colormaps.values(): 

153 cmaps = [(cmap, cmap.name), *cmaps] 

154 low, high = mappable.get_clim() 

155 mappabledata = [ 

156 ('Label', label), 

157 ('Colormap', [cmap.name] + cmaps), 

158 ('Min. value', low), 

159 ('Max. value', high), 

160 ] 

161 if hasattr(mappable, "get_interpolation"): # Images. 

162 interpolations = [ 

163 (name, name) for name in sorted(mimage.interpolations_names)] 

164 mappabledata.append(( 

165 'Interpolation', 

166 [mappable.get_interpolation(), *interpolations])) 

167 mappables.append([mappabledata, label, ""]) 

168 # Is there a scalarmappable displayed? 

169 has_sm = bool(mappables) 

170 

171 datalist = [(general, "Axes", "")] 

172 if curves: 

173 datalist.append((curves, "Curves", "")) 

174 if mappables: 

175 datalist.append((mappables, "Images, etc.", ""))

177 def apply_callback(data): 

178 """A callback to apply changes.""" 

179 orig_limits = { 

180 name: getattr(axes, f"get_{name}lim")() 

181 for name in axis_map 

182 } 

183 

184 general = data.pop(0) 

185 curves = data.pop(0) if has_curve else [] 

186 mappables = data.pop(0) if has_sm else [] 

187 if data: 

188 raise ValueError("Unexpected field") 

189 

190 title = general.pop(0) 

191 axes.set_title(title) 

192 generate_legend = general.pop() 

193 

194 for i, (name, axis) in enumerate(axis_map.items()): 

195 axis_min = general[4*i] 

196 axis_max = general[4*i + 1] 

197 axis_label = general[4*i + 2] 

198 axis_scale = general[4*i + 3] 

199 if axis.get_scale() != axis_scale: 

200 getattr(axes, f"set_{name}scale")(axis_scale) 

201 

202 axis._set_lim(axis_min, axis_max, auto=False) 

203 axis.set_label_text(axis_label) 

204 

205 # Restore the unit data 

206 axis.converter = axis_converter[name] 

207 axis.set_units(axis_units[name]) 

208 

209 # Set / Curves 

210 for index, curve in enumerate(curves): 

211 line = labeled_lines[index][1] 

212 (label, linestyle, drawstyle, linewidth, color, marker, markersize, 

213 markerfacecolor, markeredgecolor) = curve 

214 line.set_label(label) 

215 line.set_linestyle(linestyle) 

216 line.set_drawstyle(drawstyle) 

217 line.set_linewidth(linewidth) 

218 rgba = mcolors.to_rgba(color) 

219 line.set_alpha(None) 

220 line.set_color(rgba) 

221 if marker != 'none': 

222 line.set_marker(marker) 

223 line.set_markersize(markersize) 

224 line.set_markerfacecolor(markerfacecolor) 

225 line.set_markeredgecolor(markeredgecolor) 

226 

227 # Set ScalarMappables. 

228 for index, mappable_settings in enumerate(mappables): 

229 mappable = labeled_mappables[index][1] 

230 if len(mappable_settings) == 5: 

231 label, cmap, low, high, interpolation = mappable_settings 

232 mappable.set_interpolation(interpolation) 

233 elif len(mappable_settings) == 4: 

234 label, cmap, low, high = mappable_settings 

235 mappable.set_label(label) 

236 mappable.set_cmap(cm.get_cmap(cmap)) 

237 mappable.set_clim(*sorted([low, high])) 

238 

239 # re-generate legend, if checkbox is checked 

240 if generate_legend: 

241 draggable = None 

242 ncols = 1 

243 if axes.legend_ is not None: 

244 old_legend = axes.get_legend() 

245 draggable = old_legend._draggable is not None 

246 ncols = old_legend._ncols 

247 new_legend = axes.legend(ncols=ncols) 

248 if new_legend: 

249 new_legend.set_draggable(draggable) 

250 

251 # Redraw 

252 figure = axes.get_figure() 

253 figure.canvas.draw() 

254 for name in axis_map: 

255 if getattr(axes, f"get_{name}lim")() != orig_limits[name]: 

256 figure.canvas.toolbar.push_current() 

257 break

259 _formlayout.fedit( 

260 datalist, title="Figure options", parent=parent, 

261 icon=QtGui.QIcon( 

262 str(cbook._get_data_path('images', 'qt4_editor_options.svg'))), 

263 apply=apply_callback)
