192 @staticmethod 

193 def convert_path(gfx_ctx, path, transform): 

194 wxpath = gfx_ctx.CreatePath() 

195 for points, code in path.iter_segments(transform): 

196 if code == Path.MOVETO: 

197 wxpath.MoveToPoint(*points) 

198 elif code == Path.LINETO: 

199 wxpath.AddLineToPoint(*points) 

200 elif code == Path.CURVE3: 

201 wxpath.AddQuadCurveToPoint(*points) 

202 elif code == Path.CURVE4: 

203 wxpath.AddCurveToPoint(*points) 

204 elif code == Path.CLOSEPOLY: 

205 wxpath.CloseSubpath() 

206 return wxpath

275 def get_wx_font(self, s, prop): 

276 """Return a wx font. Cache font instances for efficiency.""" 

277 _log.debug("%s - get_wx_font()", type(self)) 

278 key = hash(prop) 

279 font = self.fontd.get(key) 

280 if font is not None: 

281 return font 

282 size = self.points_to_pixels(prop.get_size_in_points()) 

283 # Font colour is determined by the active wx.Pen 

284 # TODO: It may be wise to cache font information 

285 self.fontd[key] = font = wx.Font( # Cache the font and gc. 

286 pointSize=round(size), 

287 family=self.fontnames.get(prop.get_name(), wx.ROMAN), 

288 style=self.fontangles[prop.get_style()], 

289 weight=self.fontweights[prop.get_weight()]) 

290 return font

297class GraphicsContextWx(GraphicsContextBase): 

298 """ 

299 The graphics context provides the color, line styles, etc. 

300 

301 This class stores a reference to a wxMemoryDC, and a 

302 wxGraphicsContext that draws to it. Creating a wxGraphicsContext 

303 seems to be fairly heavy, so these objects are cached based on the 

304 bitmap object that is passed in. 

305 

306 The base GraphicsContext stores colors as an RGB tuple on the unit 

307 interval, e.g., (0.5, 0.0, 1.0). wxPython uses an int interval, but 

308 since wxPython colour management is rather simple, I have not chosen 

309 to implement a separate colour manager class. 

310 """ 

311 _capd = {'butt': wx.CAP_BUTT, 

312 'projecting': wx.CAP_PROJECTING, 

313 'round': wx.CAP_ROUND} 

314 

315 _joind = {'bevel': wx.JOIN_BEVEL, 

316 'miter': wx.JOIN_MITER, 

317 'round': wx.JOIN_ROUND}

327 if dc is None:
