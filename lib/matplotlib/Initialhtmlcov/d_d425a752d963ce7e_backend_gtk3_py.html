112 def _mpl_coords(self, event=None): 

113 """ 

114 Convert the position of a GTK event, or of the current cursor position 

115 if *event* is None, to Matplotlib coordinates. 

116 

117 GTK use logical pixels, but the figure is scaled to physical pixels for 

118 rendering. Transform to physical pixels so that all of the down-stream 

119 transforms work as expected. 

120 

121 Also, the origin is different and needs to be corrected. 

122 """ 

123 if event is None: 

124 window = self.get_window() 

125 t, x, y, state = window.get_device_position( 

126 window.get_display().get_device_manager().get_client_pointer()) 

127 else: 

128 x, y = event.x, event.y 

129 x = x * self.device_pixel_ratio 

130 # flip y so y=0 is bottom of canvas 

131 y = self.figure.bbox.height - y * self.device_pixel_ratio 

132 return x, y

207 def _get_key(self, event): 

208 unikey = chr(Gdk.keyval_to_unicode(event.keyval)) 

209 key = cbook._unikey_or_keysym_to_mplkey( 

210 unikey, Gdk.keyval_name(event.keyval)) 

211 mods = self._mpl_modifiers(event.state, exclude=key) 

212 if "shift" in mods and unikey.isprintable(): 

213 mods.remove("shift") 

214 return "+".join([*mods, key])

493 def _normalize_shortcut(self, key): 

494 """ 

495 Convert Matplotlib key presses to GTK+ accelerator identifiers. 

496 

497 Related to `FigureCanvasGTK3._get_key`. 

498 """ 

499 special = { 

500 'backspace': 'BackSpace', 

501 'pagedown': 'Page_Down', 

502 'pageup': 'Page_Up', 

503 'scroll_lock': 'Scroll_Lock', 

504 } 

505 

506 parts = key.split('+') 

507 mods = ['<' + mod + '>' for mod in parts[:-1]] 

508 key = parts[-1] 

509 

510 if key in special: 

511 key = special[key] 

512 elif len(key) > 1: 

513 key = key.capitalize() 

514 elif key.isupper(): 

515 mods += ['<shift>'] 

516 

517 return ''.join(mods) + key 

518 

519 def _is_valid_shortcut(self, key): 

520 """ 

521 Check for a valid shortcut to be displayed. 

522 

523 - GTK will never send 'cmd+' (see `FigureCanvasGTK3._get_key`). 

524 - The shortcut window only shows keyboard shortcuts, not mouse buttons. 

525 """ 

526 return 'cmd+' not in key and not key.startswith('MouseButton.')
