53def subprocess_run_for_testing( 

54 command: "list[str]", 

55 env: "dict[str, str]" = None, 

56 timeout: float = None, 

57 stdout=None, 

58 stderr=None, 

59 check: bool = False, 

60 text: bool = True, 

61 capture_output: bool = False 

62) -> "subprocess.Popen": 

63 """ 

64 Create and run a subprocess. 

65 

66 Thin wrapper around `subprocess.run`, intended for testing. Will 

67 mark fork() failures on Cygwin as expected failures: not a 

68 success, but not indicating a problem with the code either. 

69 

70 Parameters 

71 ---------- 

72 args : list of str 

73 env : dict[str, str] 

74 timeout : float 

75 stdout, stderr 

76 check : bool 

77 text : bool 

78 Also called ``universal_newlines`` in subprocess. I chose this 

79 name since the main effect is returning bytes (`False`) vs. str 

80 (`True`), though it also tries to normalize newlines across 

81 platforms. 

82 capture_output : bool 

83 Set stdout and stderr to subprocess.PIPE 

84 

85 Returns 

86 ------- 

87 proc : subprocess.Popen 

88 

89 See Also 

90 -------- 

91 subprocess.run 

92 

93 Raises 

94 ------ 

95 pytest.xfail 

96 If platform is Cygwin and subprocess reports a fork() failure. 

97 """ 

98 if capture_output: 

99 stdout = stderr = subprocess.PIPE 

100 try: 

101 proc = subprocess.run( 

102 command, env=env, 

103 timeout=timeout, check=check, 

104 stdout=stdout, stderr=stderr, 

105 text=text 

106 ) 

107 except BlockingIOError: 

108 if sys.platform == "cygwin": 

109 # Might want to make this more specific 

110 import pytest 

111 pytest.xfail("Fork failure") 

112 raise 

113 return proc

116def subprocess_run_helper(func, *args, timeout, extra_env=None): 

117 """ 

118 Run a function in a sub-process. 

119 

120 Parameters 

121 ---------- 

122 func : function 

123 The function to be run. It must be in a module that is importable. 

124 *args : str 

125 Any additional command line arguments to be passed in 

126 the first argument to ``subprocess.run``. 

127 extra_env : dict[str, str] 

128 Any additional environment variables to be set for the subprocess. 

129 """ 

130 target = func.__name__ 

131 module = func.__module__ 

132 proc = subprocess_run_for_testing( 

133 [ 

134 sys.executable, 

135 "-c", 

136 f"from {module} import {target}; {target}()", 

137 *args 

138 ], 

139 env={**os.environ, "SOURCE_DATE_EPOCH": "0", **(extra_env or {})}, 

140 timeout=timeout, check=True, 

141 stdout=subprocess.PIPE, 

142 stderr=subprocess.PIPE, 

143 text=True 

144 ) 

145 return proc

148def _check_for_pgf(texsystem): 

149 """ 

150 Check if a given TeX system + pgf is available 

151 

152 Parameters 

153 ---------- 

154 texsystem : str 

155 The executable name to check 

156 """ 

157 with TemporaryDirectory() as tmpdir: 

158 tex_path = Path(tmpdir, "test.tex") 

159 tex_path.write_text(r""" 

160 \documentclass{article} 

161 \usepackage{pgf} 

162 \begin{document} 

163 \typeout{pgfversion=\pgfversion} 

164 \makeatletter 

165 \@@end 

166 """, encoding="utf-8") 

167 try: 

168 subprocess.check_call( 

169 [texsystem, "-halt-on-error", str(tex_path)], cwd=tmpdir, 

170 stdout=subprocess.DEVNULL, stderr=subprocess.DEVNULL) 

171 except (OSError, subprocess.CalledProcessError): 

172 return False 

173 return True
