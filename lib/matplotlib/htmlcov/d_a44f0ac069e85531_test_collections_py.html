264def check_segments(coll, positions, linelength, lineoffset, orientation): 

265 """ 

266 Test helper checking that all values in the segment are correct, given a 

267 particular set of inputs. 

268 """ 

269 segments = coll.get_segments() 

270 if (orientation.lower() == 'horizontal' 

271 or orientation.lower() == 'none' or orientation is None): 

272 # if horizontal, the position in is in the y-axis 

273 pos1 = 1 

274 pos2 = 0 

275 elif orientation.lower() == 'vertical': 275 ↛ 280line 275 didn't jump to line 280, because the condition on line 275 was never false

276 # if vertical, the position in is in the x-axis 

277 pos1 = 0 

278 pos2 = 1 

279 else: 

280 raise ValueError("orientation must be 'horizontal' or 'vertical'") 

281 

282 # test to make sure each segment is correct 

283 for i, segment in enumerate(segments): 

284 assert segment[0, pos1] == lineoffset + linelength / 2 

285 assert segment[1, pos1] == lineoffset - linelength / 2 

286 assert segment[0, pos2] == positions[i] 

287 assert segment[1, pos2] == positions[i]
