2551 x_p = np.divide(y, norm, where=norm != 0, out=np.zeros_like(x)) 

2552 y_p = np.divide(-x, norm, where=norm != 0, out=np.ones_like(x))

2554 rangle = math.radians(15) 

2555 c = math.cos(rangle) 

2556 s = math.sin(rangle) 

2557 # construct the rotation matrices of shape (3, 3, n) 

2558 r13 = y_p * s 

2559 r32 = x_p * s 

2560 r12 = x_p * y_p * (1 - c) 

2561 Rpos = np.array( 

2562 [[c + (x_p ** 2) * (1 - c), r12, r13], 

2563 [r12, c + (y_p ** 2) * (1 - c), -r32], 

2564 [-r13, r32, np.full_like(x_p, c)]]) 

2565 # opposite rotation negates all the sin terms 

2566 Rneg = Rpos.copy() 

2567 Rneg[[0, 1, 2, 2], [2, 2, 0, 1]] *= -1 

2568 # Batch n (3, 3) x (3) matrix multiplications ((3, 3, n) x (n, 3)). 

2569 Rpos_vecs = np.einsum("ij...,...j->...i", Rpos, UVW) 

2570 Rneg_vecs = np.einsum("ij...,...j->...i", Rneg, UVW) 

2571 # Stack into (n, 2, 3) result. 

2572 return np.stack([Rpos_vecs, Rneg_vecs], axis=1)
