117def _escape_and_apply_props(s, prop): 

118 """ 

119 Generate a TeX string that renders string *s* with font properties *prop*, 

120 also applying any required escapes to *s*. 

121 """ 

122 commands = [] 

123 

124 families = {"serif": r"\rmfamily", "sans": r"\sffamily", 

125 "sans-serif": r"\sffamily", "monospace": r"\ttfamily"} 

126 family = prop.get_family()[0] 

127 if family in families: 

128 commands.append(families[family]) 

129 elif (any(font.name == family for font in fm.fontManager.ttflist) 

130 and mpl.rcParams["pgf.texsystem"] != "pdflatex"): 

131 commands.append(r"\setmainfont{%s}\rmfamily" % family) 

132 else: 

133 _log.warning("Ignoring unknown font: %s", family) 

134 

135 size = prop.get_size_in_points() 

136 commands.append(r"\fontsize{%f}{%f}" % (size, size * 1.2)) 

137 

138 styles = {"normal": r"", "italic": r"\itshape", "oblique": r"\slshape"} 

139 commands.append(styles[prop.get_style()]) 

140 

141 boldstyles = ["semibold", "demibold", "demi", "bold", "heavy", 

142 "extra bold", "black"] 

143 if prop.get_weight() in boldstyles: 

144 commands.append(r"\bfseries") 

145 

146 commands.append(r"\selectfont") 

147 return ( 

148 "{" 

149 + "".join(commands) 

150 + r"\catcode`\^=\active\def^{\ifmmode\sp\else\^{}\fi}" 

151 # It should normally be enough to set the catcode of % to 12 ("normal 

152 # character"); this works on TeXLive 2021 but not on 2018, so we just 

153 # make it active too. 

154 + r"\catcode`\%=\active\def%{\%}" 

155 + _tex_escape(s) 

156 + "}" 

157 )
