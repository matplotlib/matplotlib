67# This class handles registration of the skew-xaxes as a projection as well 

68# as setting up the appropriate transformations. It also overrides standard 

69# spines and axes instances as appropriate. 

70class SkewXAxes(Axes):

76 # Taken from Axes and modified to use our modified X-axis 

77 self.xaxis = SkewXAxis(self) 

78 self.spines.top.register_axis(self.xaxis)

91 def _set_lim_and_transforms(self): 

92 """ 

93 This is called once when the plot is created to set up all the 

94 transforms for the data, text and grids. 

95 """ 

96 rot = 30 

97 

98 # Get the standard transform setup from the Axes base class 

99 super()._set_lim_and_transforms() 

100 

101 # Need to put the skew in the middle, after the scale and limits, 

102 # but before the transAxes. This way, the skew is done in Axes 

103 # coordinates thus performing the transform around the proper origin 

104 # We keep the pre-transAxes transform around for other users, like the 

105 # spines for finding bounds 

106 self.transDataToAxes = (self.transScale + 

107 (self.transLimits + 

108 transforms.Affine2D().skew_deg(rot, 0))) 

109 

110 # Create the full transform from Data to Pixels 

111 self.transData = self.transDataToAxes + self.transAxes 

112 

113 # Blended transforms like this need to have the skewing applied using 

114 # both axes, in axes coords like before. 

115 self._xaxis_transform = (transforms.blended_transform_factory( 

116 self.transScale + self.transLimits, 

117 transforms.IdentityTransform()) + 

118 transforms.Affine2D().skew_deg(rot, 0)) + self.transAxes 

119 

120 @property 

121 def lower_xlim(self): 

122 return self.axes.viewLim.intervalx 

123 

124 @property 

125 def upper_xlim(self): 

126 pts = [[0., 1.], [1., 1.]] 

127 return self.transDataToAxes.inverted().transform(pts)[:, 0] 

128 

129 

130# Now register the projection with matplotlib so the user can select 

131# it. 

132register_projection(SkewXAxes)
