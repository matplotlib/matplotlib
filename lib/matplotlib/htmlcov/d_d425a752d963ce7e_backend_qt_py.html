101def _create_qApp(): 

102 app = QtWidgets.QApplication.instance() 

103 

104 # Create a new QApplication and configure it if none exists yet, as only 

105 # one QApplication can exist at a time. 

106 if app is None: 

107 # display_is_valid returns False only if on Linux and neither X11 

108 # nor Wayland display can be opened. 

109 if not mpl._c_internal_utils.display_is_valid(): 

110 raise RuntimeError('Invalid DISPLAY variable') 

111 

112 # Check to make sure a QApplication from a different major version 

113 # of Qt is not instantiated in the process 

114 if QT_API in {'PyQt6', 'PySide6'}: 

115 other_bindings = ('PyQt5', 'PySide2') 

116 elif QT_API in {'PyQt5', 'PySide2'}: 

117 other_bindings = ('PyQt6', 'PySide6') 

118 else: 

119 raise RuntimeError("Should never be here") 

120 

121 for binding in other_bindings: 

122 mod = sys.modules.get(f'{binding}.QtWidgets') 

123 if mod is not None and mod.QApplication.instance() is not None: 

124 other_core = sys.modules.get(f'{binding}.QtCore') 

125 _api.warn_external( 

126 f'Matplotlib is using {QT_API} which wraps ' 

127 f'{QtCore.qVersion()} however an instantiated ' 

128 f'QApplication from {binding} which wraps ' 

129 f'{other_core.qVersion()} exists. Mixing Qt major ' 

130 'versions may not work as expected.' 

131 ) 

132 break 

133 try: 

134 QtWidgets.QApplication.setAttribute( 

135 QtCore.Qt.AA_EnableHighDpiScaling) 

136 except AttributeError: # Only for Qt>=5.6, <6. 

137 pass 

138 try: 

139 QtWidgets.QApplication.setHighDpiScaleFactorRoundingPolicy( 

140 QtCore.Qt.HighDpiScaleFactorRoundingPolicy.PassThrough) 

141 except AttributeError: # Only for Qt>=5.14. 

142 pass 

143 app = QtWidgets.QApplication(["matplotlib"]) 

144 if sys.platform == "darwin": 

145 image = str(cbook._get_data_path('images/matplotlib.svg')) 

146 icon = QtGui.QIcon(image) 

147 app.setWindowIcon(icon) 

148 app.lastWindowClosed.connect(app.quit) 

149 cbook._setup_new_guiapp() 

150 

151 try: 

152 app.setAttribute(QtCore.Qt.AA_UseHighDpiPixmaps) # Only for Qt<6. 

153 except AttributeError: 

154 pass

247 def mouseEventCoords(self, pos=None): 

248 """ 

249 Calculate mouse coordinates in physical pixels. 

250 

251 Qt uses logical pixels, but the figure is scaled to physical 

252 pixels for rendering. Transform to physical pixels so that 

253 all of the down-stream transforms work as expected. 

254 

255 Also, the origin is different and needs to be corrected. 

256 """ 

257 if pos is None: 

258 pos = self.mapFromGlobal(QtGui.QCursor.pos()) 

259 elif hasattr(pos, "position"): # qt6 QtGui.QEvent 

260 pos = pos.position() 

261 elif hasattr(pos, "pos"): # qt5 QtCore.QEvent 

262 pos = pos.pos() 

263 # (otherwise, it's already a QPoint) 

264 x = pos.x() 

265 # flip y so y=0 is bottom of canvas 

266 y = self.figure.bbox.height / self.device_pixel_ratio - pos.y() 

267 return x * self.device_pixel_ratio, y * self.device_pixel_ratio

370 def _mpl_modifiers(modifiers=None, *, exclude=None): 

371 if modifiers is None: 

372 modifiers = QtWidgets.QApplication.instance().keyboardModifiers() 

373 modifiers = _to_int(modifiers) 

374 # get names of the pressed modifier keys 

375 # 'control' is named 'control' when a standalone key, but 'ctrl' when a 

376 # modifier 

377 # bit twiddling to pick out modifier keys from modifiers bitmask, 

378 # if exclude is a MODIFIER, it should not be duplicated in mods 

379 return [SPECIAL_KEYS[key].replace('control', 'ctrl') 

380 for mask, key in _MODIFIER_KEYS 

381 if exclude != key and modifiers & mask]

383 def _get_key(self, event): 

384 event_key = event.key() 

385 mods = self._mpl_modifiers(exclude=event_key) 

386 try: 

387 # for certain keys (enter, left, backspace, etc) use a word for the 

388 # key, rather than Unicode 

389 key = SPECIAL_KEYS[event_key] 

390 except KeyError: 

391 # Unicode defines code points up to 0x10ffff (sys.maxunicode) 

392 # QT will use Key_Codes larger than that for keyboard keys that are 

393 # not Unicode characters (like multimedia keys) 

394 # skip these 

395 # if you really want them, you should add them to SPECIAL_KEYS 

396 if event_key > sys.maxunicode: 

397 return None 

398 

399 key = chr(event_key) 

400 # qt delivers capitalized letters. fix capitalization 

401 # note that capslock is ignored 

402 if 'shift' in mods: 

403 mods.remove('shift') 

404 else: 

405 key = key.lower() 

406 

407 return '+'.join(mods + [key])

681 def _icon(self, name): 

682 """ 

683 Construct a `.QIcon` from an image file *name*, including the extension 

684 and relative to Matplotlib's "images" data directory. 

685 """ 

686 # use a high-resolution icon with suffix '_large' if available 

687 # note: user-provided icons may not have '_large' versions 

688 path_regular = cbook._get_data_path('images', name) 

689 path_large = path_regular.with_name( 

690 path_regular.name.replace('.png', '_large.png')) 

691 filename = str(path_large if path_large.exists() else path_regular) 

692 

693 pm = QtGui.QPixmap(filename) 

694 pm.setDevicePixelRatio( 

695 self.devicePixelRatioF() or 1) # rarely, devicePixelRatioF=0 

696 if self.palette().color(self.backgroundRole()).value() < 128: 

697 icon_color = self.palette().color(self.foregroundRole()) 

698 mask = pm.createMaskFromColor( 

699 QtGui.QColor('black'), 

700 _enum("QtCore.Qt.MaskMode").MaskOutColor) 

701 pm.fill(icon_color) 

702 pm.setMask(mask) 

703 return QtGui.QIcon(pm)

794 filters, selectedFilter) 

795 if fname: 

796 # Save dir for next time, unless empty str (i.e., use cwd). 

797 if startpath != "": 

798 mpl.rcParams['savefig.directory'] = os.path.dirname(fname) 

799 try: 

800 self.canvas.figure.savefig(fname) 

801 except Exception as e: 

802 QtWidgets.QMessageBox.critical( 

803 self, "Error saving file", str(e), 

804 _enum("QtWidgets.QMessageBox.StandardButton").Ok, 

805 _enum("QtWidgets.QMessageBox.StandardButton").NoButton)

816class SubplotToolQt(QtWidgets.QDialog): 

817 def __init__(self, targetfig, parent): 

818 super().__init__() 

819 self.setWindowIcon(QtGui.QIcon( 

820 str(cbook._get_data_path("images/matplotlib.png")))) 

821 self.setObjectName("SubplotTool") 

822 self._spinboxes = {} 

823 main_layout = QtWidgets.QHBoxLayout() 

824 self.setLayout(main_layout) 

825 for group, spinboxes, buttons in [ 

826 ("Borders", 

827 ["top", "bottom", "left", "right"], 

828 [("Export values", self._export_values)]), 

829 ("Spacings", 

830 ["hspace", "wspace"], 

831 [("Tight layout", self._tight_layout), 

832 ("Reset", self._reset), 

833 ("Close", self.close)])]: 

834 layout = QtWidgets.QVBoxLayout() 

835 main_layout.addLayout(layout) 

836 box = QtWidgets.QGroupBox(group) 

837 layout.addWidget(box) 

838 inner = QtWidgets.QFormLayout(box) 

839 for name in spinboxes: 

840 self._spinboxes[name] = spinbox = QtWidgets.QDoubleSpinBox() 

841 spinbox.setRange(0, 1) 

842 spinbox.setDecimals(3) 

843 spinbox.setSingleStep(0.005) 

844 spinbox.setKeyboardTracking(False) 

845 spinbox.valueChanged.connect(self._on_value_changed) 

846 inner.addRow(name, spinbox) 

847 layout.addStretch(1) 

848 for name, method in buttons: 

849 button = QtWidgets.QPushButton(name) 

850 # Don't trigger on <enter>, which is used to input values. 

851 button.setAutoDefault(False) 

852 button.clicked.connect(method) 

853 layout.addWidget(button) 

854 if name == "Close": 

855 button.setFocus() 

856 self._figure = targetfig 

857 self._defaults = {} 

858 self._export_values_dialog = None 

859 self.update_from_current_subplotpars() 

860 

861 def update_from_current_subplotpars(self): 

862 self._defaults = {spinbox: getattr(self._figure.subplotpars, name) 

863 for name, spinbox in self._spinboxes.items()} 

864 self._reset() # Set spinbox current values without triggering signals. 

865 

866 def _export_values(self): 

867 # Explicitly round to 3 decimals (which is also the spinbox precision) 

868 # to avoid numbers of the form 0.100...001. 

869 self._export_values_dialog = QtWidgets.QDialog() 

870 layout = QtWidgets.QVBoxLayout() 

871 self._export_values_dialog.setLayout(layout) 

872 text = QtWidgets.QPlainTextEdit() 

873 text.setReadOnly(True) 

874 layout.addWidget(text) 

875 text.setPlainText( 

876 ",\n".join(f"{attr}={spinbox.value():.3}" 

877 for attr, spinbox in self._spinboxes.items())) 

878 # Adjust the height of the text widget to fit the whole text, plus 

879 # some padding. 

880 size = text.maximumSize() 

881 size.setHeight( 

882 QtGui.QFontMetrics(text.document().defaultFont()) 

883 .size(0, text.toPlainText()).height() + 20) 

884 text.setMaximumSize(size) 

885 self._export_values_dialog.show() 

886 

887 def _on_value_changed(self): 

888 spinboxes = self._spinboxes 

889 # Set all mins and maxes, so that this can also be used in _reset(). 

890 for lower, higher in [("bottom", "top"), ("left", "right")]: 

891 spinboxes[higher].setMinimum(spinboxes[lower].value() + .001) 

892 spinboxes[lower].setMaximum(spinboxes[higher].value() - .001) 

893 self._figure.subplots_adjust( 

894 **{attr: spinbox.value() for attr, spinbox in spinboxes.items()}) 

895 self._figure.canvas.draw_idle() 

896 

897 def _tight_layout(self): 

898 self._figure.tight_layout() 

899 for attr, spinbox in self._spinboxes.items(): 

900 spinbox.blockSignals(True) 

901 spinbox.setValue(vars(self._figure.subplotpars)[attr]) 

902 spinbox.blockSignals(False) 

903 self._figure.canvas.draw_idle() 

904 

905 def _reset(self): 

906 for spinbox, value in self._defaults.items(): 

907 spinbox.setRange(0, 1) 

908 spinbox.blockSignals(True) 

909 spinbox.setValue(value) 

910 spinbox.blockSignals(False) 

911 self._on_value_changed()

933 def add_toolitem( 

934 self, name, group, position, image_file, description, toggle): 

935 

936 button = QtWidgets.QToolButton(self) 

937 if image_file: 

938 button.setIcon(NavigationToolbar2QT._icon(self, image_file)) 

939 button.setText(name) 

940 if description: 

941 button.setToolTip(description)

943 def handler(): 

944 self.trigger_tool(name) 

945 if toggle: 

946 button.setCheckable(True) 

947 button.toggled.connect(handler) 

948 else: 

949 button.clicked.connect(handler) 

950 

951 self._toolitems.setdefault(name, []) 

952 self._add_to_group(group, name, button, position) 

953 self._toolitems[name].append((button, handler)) 

954 

955 def _add_to_group(self, group, name, button, position): 

956 gr = self._groups.get(group, []) 

957 if not gr: 

958 sep = self.insertSeparator(self._message_action) 

959 gr.append(sep) 

960 before = gr[position] 

961 widget = self.insertWidget(before, button) 

962 gr.insert(position, widget) 

963 self._groups[group] = gr 

964 

965 def toggle_toolitem(self, name, toggled): 

966 if name not in self._toolitems: 

967 return 

968 for button, handler in self._toolitems[name]: 

969 button.toggled.disconnect(handler) 

970 button.setChecked(toggled) 

971 button.toggled.connect(handler) 

972 

973 def remove_toolitem(self, name): 

974 for button, handler in self._toolitems[name]: 

975 button.setParent(None) 

976 del self._toolitems[name]

979 self.widgetForAction(self._message_action).setText(s)
