655@pytest.mark.parametrize('locked_element', ['x0', 'y0', 'x1', 'y1']) 

656def test_lockable_bbox(locked_element): 

657 other_elements = ['x0', 'y0', 'x1', 'y1'] 

658 other_elements.remove(locked_element) 

659 

660 orig = mtransforms.Bbox.unit() 

661 locked = mtransforms.LockableBbox(orig, **{locked_element: 2}) 

662 

663 # LockableBbox should keep its locked element as specified in __init__. 

664 assert getattr(locked, locked_element) == 2 

665 assert getattr(locked, 'locked_' + locked_element) == 2 

666 for elem in other_elements: 

667 assert getattr(locked, elem) == getattr(orig, elem) 

668 

669 # Changing underlying Bbox should update everything but locked element. 

670 orig.set_points(orig.get_points() + 10) 

671 assert getattr(locked, locked_element) == 2 

672 assert getattr(locked, 'locked_' + locked_element) == 2 

673 for elem in other_elements: 

674 assert getattr(locked, elem) == getattr(orig, elem) 

675 

676 # Unlocking element should revert values back to the underlying Bbox. 

677 setattr(locked, 'locked_' + locked_element, None) 

678 assert getattr(locked, 'locked_' + locked_element) is None 

679 assert np.all(orig.get_points() == locked.get_points()) 

680 

681 # Relocking an element should change its value, but not others. 

682 setattr(locked, 'locked_' + locked_element, 3) 

683 assert getattr(locked, locked_element) == 3 

684 assert getattr(locked, 'locked_' + locked_element) == 3 

685 for elem in other_elements: 

686 assert getattr(locked, elem) == getattr(orig, elem)

689def test_copy(): 

690 a = mtransforms.Affine2D() 

691 b = mtransforms.Affine2D() 

692 s = a + b 

693 # Updating a dependee should invalidate a copy of the dependent. 

694 s.get_matrix() # resolve it. 

695 s1 = copy.copy(s) 

696 assert not s._invalid and not s1._invalid 

697 a.translate(1, 2) 

698 assert s._invalid and s1._invalid

702 b1 = copy.copy(b) 

703 b1.translate(3, 4) 

704 assert not s._invalid 

705 assert (s.get_matrix() == a.get_matrix()).all()
