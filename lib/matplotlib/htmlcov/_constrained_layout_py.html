1""" 

2Adjust subplot layouts so that there are no overlapping axes or axes 

3decorations. All axes decorations are dealt with (labels, ticks, titles, 

4ticklabels) and some dependent artists are also dealt with (colorbar, 

5suptitle). 

6 

7Layout is done via `~matplotlib.gridspec`, with one constraint per gridspec, 

8so it is possible to have overlapping axes if the gridspecs overlap (i.e. 

9using `~matplotlib.gridspec.GridSpecFromSubplotSpec`). Axes placed using 

10``figure.subplots()`` or ``figure.add_subplots()`` will participate in the 

11layout. Axes manually placed via ``figure.add_axes()`` will not. 

12 

13See Tutorial: :doc:`/tutorials/intermediate/constrainedlayout_guide` 

14 

15General idea: 

16------------- 

17 

18First, a figure has a gridspec that divides the figure into nrows and ncols, 

19with heights and widths set by ``height_ratios`` and ``width_ratios``, 

20often just set to 1 for an equal grid. 

21 

22Subplotspecs that are derived from this gridspec can contain either a 

23``SubPanel``, a ``GridSpecFromSubplotSpec``, or an ``Axes``. The ``SubPanel`` 

24and ``GridSpecFromSubplotSpec`` are dealt with recursively and each contain an 

25analogous layout. 

26 

27Each ``GridSpec`` has a ``_layoutgrid`` attached to it. The ``_layoutgrid`` 

28has the same logical layout as the ``GridSpec``. Each row of the grid spec 

29has a top and bottom "margin" and each column has a left and right "margin". 

30The "inner" height of each row is constrained to be the same (or as modified 

31by ``height_ratio``), and the "inner" width of each column is 

32constrained to be the same (as modified by ``width_ratio``), where "inner" 

33is the width or height of each column/row minus the size of the margins. 

34 

35Then the size of the margins for each row and column are determined as the 

36max width of the decorators on each axes that has decorators in that margin. 

37For instance, a normal axes would have a left margin that includes the 

38left ticklabels, and the ylabel if it exists. The right margin may include a 

39colorbar, the bottom margin the xaxis decorations, and the top margin the 

40title. 

41 

42With these constraints, the solver then finds appropriate bounds for the 

43columns and rows. It's possible that the margins take up the whole figure, 

44in which case the algorithm is not applied and a warning is raised. 

45 

46See the tutorial doc:`/tutorials/intermediate/constrainedlayout_guide` 

47for more discussion of the algorithm with examples. 

48"""

300def get_margin_from_padding(obj, *, w_pad=0, h_pad=0, 

301 hspace=0, wspace=0): 

302 

303 ss = obj._subplotspec 

304 gs = ss.get_gridspec() 

305 

306 if hasattr(gs, 'hspace'): 

307 _hspace = (gs.hspace if gs.hspace is not None else hspace) 

308 _wspace = (gs.wspace if gs.wspace is not None else wspace) 

309 else: 

310 _hspace = (gs._hspace if gs._hspace is not None else hspace) 

311 _wspace = (gs._wspace if gs._wspace is not None else wspace) 

312 

313 _wspace = _wspace / 2 

314 _hspace = _hspace / 2 

315 

316 nrows, ncols = gs.get_geometry() 

317 # there are two margins for each direction. The "cb" 

318 # margins are for pads and colorbars, the non-"cb" are 

319 # for the axes decorations (labels etc). 

320 margin = {'leftcb': w_pad, 'rightcb': w_pad, 

321 'bottomcb': h_pad, 'topcb': h_pad, 

322 'left': 0, 'right': 0, 

323 'top': 0, 'bottom': 0} 

324 if _wspace / ncols > w_pad: 

325 if ss.colspan.start > 0: 

326 margin['leftcb'] = _wspace / ncols 

327 if ss.colspan.stop < ncols: 

328 margin['rightcb'] = _wspace / ncols 

329 if _hspace / nrows > h_pad: 

330 if ss.rowspan.stop < nrows: 

331 margin['bottomcb'] = _hspace / nrows 

332 if ss.rowspan.start > 0: 

333 margin['topcb'] = _hspace / nrows 

334 

335 return margin

480def match_submerged_margins(layoutgrids, fig): 

481 """ 

482 Make the margins that are submerged inside an Axes the same size. 

483 

484 This allows axes that span two columns (or rows) that are offset 

485 from one another to have the same size. 

486 

487 This gives the proper layout for something like:: 

488 fig = plt.figure(constrained_layout=True) 

489 axs = fig.subplot_mosaic("AAAB\nCCDD") 

490 

491 Without this routine, the axes D will be wider than C, because the 

492 margin width between the two columns in C has no width by default, 

493 whereas the margins between the two columns of D are set by the 

494 width of the margin between A and B. However, obviously the user would 

495 like C and D to be the same size, so we need to add constraints to these 

496 "submerged" margins. 

497 

498 This routine makes all the interior margins the same, and the spacing 

499 between the three columns in A and the two column in C are all set to the 

500 margins between the two columns of D. 

501 

502 See test_constrained_layout::test_constrained_layout12 for an example. 

503 """ 

504 

505 for sfig in fig.subfigs: 

506 match_submerged_margins(layoutgrids, sfig) 

507 

508 axs = [a for a in fig.get_axes() 

509 if a.get_subplotspec() is not None and a.get_in_layout()] 

510 

511 for ax1 in axs: 

512 ss1 = ax1.get_subplotspec() 

513 if ss1.get_gridspec() not in layoutgrids: 513 ↛ 514line 513 didn't jump to line 514, because the condition on line 513 was never true

514 axs.remove(ax1) 

515 continue 

516 lg1 = layoutgrids[ss1.get_gridspec()] 

517 

518 # interior columns: 

519 if len(ss1.colspan) > 1: 

520 maxsubl = np.max( 

521 lg1.margin_vals['left'][ss1.colspan[1:]] + 

522 lg1.margin_vals['leftcb'][ss1.colspan[1:]] 

523 ) 

524 maxsubr = np.max( 

525 lg1.margin_vals['right'][ss1.colspan[:-1]] + 

526 lg1.margin_vals['rightcb'][ss1.colspan[:-1]] 

527 ) 

528 for ax2 in axs: 

529 ss2 = ax2.get_subplotspec() 

530 lg2 = layoutgrids[ss2.get_gridspec()] 

531 if lg2 is not None and len(ss2.colspan) > 1: 

532 maxsubl2 = np.max( 

533 lg2.margin_vals['left'][ss2.colspan[1:]] + 

534 lg2.margin_vals['leftcb'][ss2.colspan[1:]]) 

535 if maxsubl2 > maxsubl: 

536 maxsubl = maxsubl2 

537 maxsubr2 = np.max( 

538 lg2.margin_vals['right'][ss2.colspan[:-1]] + 

539 lg2.margin_vals['rightcb'][ss2.colspan[:-1]]) 

540 if maxsubr2 > maxsubr: 

541 maxsubr = maxsubr2 

542 for i in ss1.colspan[1:]: 

543 lg1.edit_margin_min('left', maxsubl, cell=i) 

544 for i in ss1.colspan[:-1]: 

545 lg1.edit_margin_min('right', maxsubr, cell=i) 

546 

547 # interior rows: 

548 if len(ss1.rowspan) > 1: 

549 maxsubt = np.max( 

550 lg1.margin_vals['top'][ss1.rowspan[1:]] + 

551 lg1.margin_vals['topcb'][ss1.rowspan[1:]] 

552 ) 

553 maxsubb = np.max( 

554 lg1.margin_vals['bottom'][ss1.rowspan[:-1]] + 

555 lg1.margin_vals['bottomcb'][ss1.rowspan[:-1]] 

556 ) 

557 

558 for ax2 in axs: 

559 ss2 = ax2.get_subplotspec() 

560 lg2 = layoutgrids[ss2.get_gridspec()] 

561 if lg2 is not None: 561 ↛ 558line 561 didn't jump to line 558, because the condition on line 561 was never false

562 if len(ss2.rowspan) > 1: 

563 maxsubt = np.max([np.max( 

564 lg2.margin_vals['top'][ss2.rowspan[1:]] + 

565 lg2.margin_vals['topcb'][ss2.rowspan[1:]] 

566 ), maxsubt]) 

567 maxsubb = np.max([np.max( 

568 lg2.margin_vals['bottom'][ss2.rowspan[:-1]] + 

569 lg2.margin_vals['bottomcb'][ss2.rowspan[:-1]] 

570 ), maxsubb]) 

571 for i in ss1.rowspan[1:]: 

572 lg1.edit_margin_min('top', maxsubt, cell=i) 

573 for i in ss1.rowspan[:-1]: 

574 lg1.edit_margin_min('bottom', maxsubb, cell=i)

577def get_cb_parent_spans(cbax): 

578 """ 

579 Figure out which subplotspecs this colorbar belongs to: 

580 """ 

581 rowstart = np.inf 

582 rowstop = -np.inf 

583 colstart = np.inf 

584 colstop = -np.inf 

585 for parent in cbax._colorbar_info['parents']: 

586 ss = parent.get_subplotspec() 

587 rowstart = min(ss.rowspan.start, rowstart) 

588 rowstop = max(ss.rowspan.stop, rowstop) 

589 colstart = min(ss.colspan.start, colstart) 

590 colstop = max(ss.colspan.stop, colstop) 

591 

592 rowspan = range(rowstart, rowstop) 

593 colspan = range(colstart, colstop) 

594 return rowspan, colspan

597def get_pos_and_bbox(ax, renderer): 

598 """ 

599 Get the position and the bbox for the axes. 

600 

601 Parameters 

602 ---------- 

603 ax 

604 renderer 

605 

606 Returns 

607 ------- 

608 pos : Bbox 

609 Position in figure coordinates. 

610 bbox : Bbox 

611 Tight bounding box in figure coordinates. 

612 """ 

613 fig = ax.figure 

614 pos = ax.get_position(original=True) 

615 # pos is in panel co-ords, but we need in figure for the layout 

616 pos = pos.transformed(fig.transSubfigure - fig.transFigure) 

617 tightbbox = martist._get_tightbbox_for_layout_only(ax, renderer) 

618 if tightbbox is None: 

619 bbox = pos 

620 else: 

621 bbox = tightbbox.transformed(fig.transFigure.inverted()) 

622 return pos, bbox

625def reposition_axes(layoutgrids, fig, renderer, *, 

626 w_pad=0, h_pad=0, hspace=0, wspace=0): 

627 """ 

628 Reposition all the axes based on the new inner bounding box. 

629 """ 

630 trans_fig_to_subfig = fig.transFigure - fig.transSubfigure 

631 for sfig in fig.subfigs: 

632 bbox = layoutgrids[sfig].get_outer_bbox() 

633 sfig._redo_transform_rel_fig( 

634 bbox=bbox.transformed(trans_fig_to_subfig)) 

635 reposition_axes(layoutgrids, sfig, renderer, 

636 w_pad=w_pad, h_pad=h_pad, 

637 wspace=wspace, hspace=hspace) 

638 

639 for ax in fig._localaxes: 

640 if ax.get_subplotspec() is None or not ax.get_in_layout(): 

641 continue 

642 

643 # grid bbox is in Figure coordinates, but we specify in panel 

644 # coordinates... 

645 ss = ax.get_subplotspec() 

646 gs = ss.get_gridspec() 

647 if gs not in layoutgrids: 647 ↛ 648line 647 didn't jump to line 648, because the condition on line 647 was never true

648 return 

649 

650 bbox = layoutgrids[gs].get_inner_bbox(rows=ss.rowspan, 

651 cols=ss.colspan) 

652 

653 # transform from figure to panel for set_position: 

654 newbbox = trans_fig_to_subfig.transform_bbox(bbox) 

655 ax._set_position(newbbox) 

656 

657 # move the colorbars: 

658 # we need to keep track of oldw and oldh if there is more than 

659 # one colorbar: 

660 offset = {'left': 0, 'right': 0, 'bottom': 0, 'top': 0} 

661 for nn, cbax in enumerate(ax._colorbars[::-1]): 

662 if ax == cbax._colorbar_info['parents'][0]: 

663 reposition_colorbar(layoutgrids, cbax, renderer, 

664 offset=offset)

667def reposition_colorbar(layoutgrids, cbax, renderer, *, offset=None): 

668 """ 

669 Place the colorbar in its new place. 

670 

671 Parameters 

672 ---------- 

673 cbax : Axes 

674 Axes for the colorbar 

675 

676 renderer : 

677 w_pad, h_pad : float 

678 width and height padding (in fraction of figure) 

679 hspace, wspace : float 

680 width and height padding as fraction of figure size divided by 

681 number of columns or rows 

682 margin : array-like 

683 offset the colorbar needs to be pushed to in order to 

684 account for multiple colorbars 

685 """ 

686 

687 parents = cbax._colorbar_info['parents'] 

688 gs = parents[0].get_gridspec() 

689 fig = cbax.figure 

690 trans_fig_to_subfig = fig.transFigure - fig.transSubfigure 

691 

692 cb_rspans, cb_cspans = get_cb_parent_spans(cbax) 

693 bboxparent = layoutgrids[gs].get_bbox_for_cb(rows=cb_rspans, 

694 cols=cb_cspans) 

695 pb = layoutgrids[gs].get_inner_bbox(rows=cb_rspans, cols=cb_cspans) 

696 

697 location = cbax._colorbar_info['location'] 

698 anchor = cbax._colorbar_info['anchor'] 

699 fraction = cbax._colorbar_info['fraction'] 

700 aspect = cbax._colorbar_info['aspect'] 

701 shrink = cbax._colorbar_info['shrink'] 

702 

703 cbpos, cbbbox = get_pos_and_bbox(cbax, renderer) 

704 

705 # Colorbar gets put at extreme edge of outer bbox of the subplotspec 

706 # It needs to be moved in by: 1) a pad 2) its "margin" 3) by 

707 # any colorbars already added at this location: 

708 cbpad = colorbar_get_pad(layoutgrids, cbax) 

709 if location in ('left', 'right'): 

710 # fraction and shrink are fractions of parent 

711 pbcb = pb.shrunk(fraction, shrink).anchored(anchor, pb) 

712 # The colorbar is at the left side of the parent. Need 

713 # to translate to right (or left) 

714 if location == 'right': 

715 lmargin = cbpos.x0 - cbbbox.x0 

716 dx = bboxparent.x1 - pbcb.x0 + offset['right'] 

717 dx += cbpad + lmargin 

718 offset['right'] += cbbbox.width + cbpad 

719 pbcb = pbcb.translated(dx, 0) 

720 else: 

721 lmargin = cbpos.x0 - cbbbox.x0 

722 dx = bboxparent.x0 - pbcb.x0 # edge of parent 

723 dx += -cbbbox.width - cbpad + lmargin - offset['left'] 

724 offset['left'] += cbbbox.width + cbpad 

725 pbcb = pbcb.translated(dx, 0) 

726 else: # horizontal axes: 

727 pbcb = pb.shrunk(shrink, fraction).anchored(anchor, pb) 

728 if location == 'top': 

729 bmargin = cbpos.y0 - cbbbox.y0 

730 dy = bboxparent.y1 - pbcb.y0 + offset['top'] 

731 dy += cbpad + bmargin 

732 offset['top'] += cbbbox.height + cbpad 

733 pbcb = pbcb.translated(0, dy) 

734 else: 

735 bmargin = cbpos.y0 - cbbbox.y0 

736 dy = bboxparent.y0 - pbcb.y0 

737 dy += -cbbbox.height - cbpad + bmargin - offset['bottom'] 

738 offset['bottom'] += cbbbox.height + cbpad 

739 pbcb = pbcb.translated(0, dy) 

740 

741 pbcb = trans_fig_to_subfig.transform_bbox(pbcb) 

742 cbax.set_transform(fig.transSubfigure) 

743 cbax._set_position(pbcb) 

744 cbax.set_anchor(anchor) 

745 if location in ['bottom', 'top']: 

746 aspect = 1 / aspect 

747 cbax.set_box_aspect(aspect) 

748 cbax.set_aspect('auto') 

749 return offset

752def reset_margins(layoutgrids, fig): 

753 """ 

754 Reset the margins in the layoutboxes of fig. 

755 

756 Margins are usually set as a minimum, so if the figure gets smaller 

757 the minimum needs to be zero in order for it to grow again. 

758 """ 

759 for sfig in fig.subfigs: 

760 reset_margins(layoutgrids, sfig) 

761 for ax in fig.axes: 

762 if ax.get_in_layout(): 

763 gs = ax.get_gridspec() 

764 if gs in layoutgrids: # also implies gs is not None. 

765 layoutgrids[gs].reset_margins() 

766 layoutgrids[fig].reset_margins()

769def colorbar_get_pad(layoutgrids, cax): 

770 parents = cax._colorbar_info['parents'] 

771 gs = parents[0].get_gridspec() 

772 

773 cb_rspans, cb_cspans = get_cb_parent_spans(cax) 

774 bboxouter = layoutgrids[gs].get_inner_bbox(rows=cb_rspans, cols=cb_cspans) 

775 

776 if cax._colorbar_info['location'] in ['right', 'left']: 

777 size = bboxouter.width 

778 else: 

779 size = bboxouter.height 

780 

781 return cax._colorbar_info['pad'] * size
