41 def _extend_path(self, path, mutation_size=10): 

42 """ 

43 Extend the path to make a room for drawing arrow. 

44 """ 

45 (x0, y0), (x1, y1) = path.vertices[-2:] 

46 theta = math.atan2(y1 - y0, x1 - x0) 

47 x2 = x1 + math.cos(theta) * mutation_size 

48 y2 = y1 + math.sin(theta) * mutation_size 

49 if path.codes is None: 

50 return Path(np.concatenate([path.vertices, [[x2, y2]]])) 

51 else: 

52 return Path(np.concatenate([path.vertices, [[x2, y2]]]), 

53 np.concatenate([path.codes, [Path.LINETO]]))
