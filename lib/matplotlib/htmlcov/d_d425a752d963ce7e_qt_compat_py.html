176@contextlib.contextmanager 

177def _maybe_allow_interrupt(qapp): 

178 """ 

179 This manager allows to terminate a plot by sending a SIGINT. It is 

180 necessary because the running Qt backend prevents Python interpreter to 

181 run and process signals (i.e., to raise KeyboardInterrupt exception). To 

182 solve this one needs to somehow wake up the interpreter and make it close 

183 the plot window. We do this by using the signal.set_wakeup_fd() function 

184 which organizes a write of the signal number into a socketpair connected 

185 to the QSocketNotifier (since it is part of the Qt backend, it can react 

186 to that write event). Afterwards, the Qt handler empties the socketpair 

187 by a recv() command to re-arm it (we need this if a signal different from 

188 SIGINT was caught by set_wakeup_fd() and we shall continue waiting). If 

189 the SIGINT was caught indeed, after exiting the on_signal() function the 

190 interpreter reacts to the SIGINT according to the handle() function which 

191 had been set up by a signal.signal() call: it causes the qt_object to 

192 exit by calling its quit() method. Finally, we call the old SIGINT 

193 handler with the same arguments that were given to our custom handle() 

194 handler. 

195 

196 We do this only if the old handler for SIGINT was not None, which means 

197 that a non-python handler was installed, i.e. in Julia, and not SIG_IGN 

198 which means we should ignore the interrupts. 

199 """ 

200 old_sigint_handler = signal.getsignal(signal.SIGINT) 

201 handler_args = None 

202 skip = False 

203 if old_sigint_handler in (None, signal.SIG_IGN, signal.SIG_DFL): 

204 skip = True 

205 else: 

206 wsock, rsock = socket.socketpair() 

207 wsock.setblocking(False) 

208 old_wakeup_fd = signal.set_wakeup_fd(wsock.fileno()) 

209 sn = QtCore.QSocketNotifier( 

210 rsock.fileno(), _enum('QtCore.QSocketNotifier.Type').Read 

211 ) 

212 

213 # We do not actually care about this value other than running some 

214 # Python code to ensure that the interpreter has a chance to handle the 

215 # signal in Python land. We also need to drain the socket because it 

216 # will be written to as part of the wakeup! There are some cases where 

217 # this may fire too soon / more than once on Windows so we should be 

218 # forgiving about reading an empty socket. 

219 rsock.setblocking(False) 

220 # Clear the socket to re-arm the notifier. 

221 @sn.activated.connect 

222 def _may_clear_sock(*args): 

223 try: 

224 rsock.recv(1) 

225 except BlockingIOError: 

226 pass

229 nonlocal handler_args 

230 handler_args = args 

231 qapp.quit() 

232 

233 signal.signal(signal.SIGINT, handle)

235 yield
